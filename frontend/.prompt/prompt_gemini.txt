## 🚀 SPRINT 1: Infraestrutura de Roteamento SPA Vanilla

**Objetivo:** Configurar o ciclo de vida da aplicação e o roteador Single Page Application usando a History API nativa.

### Instruções de Implementação:

1. Crie o arquivo `src/app/router/navigate.js` para disparar eventos de navegação global.


2. Crie o arquivo `src/app/router/matchRoute.js` para processar parâmetros dinâmicos de URL (ex: extrair o valor de `:slug` em `/game/:slug`).


3. Crie o arquivo `src/app/router/guards.js` para gerenciar a proteção de rotas privadas.


4. Crie o arquivo `src/app/router/routes.js` contendo o mapa de rotas e o carregamento sob demanda (*Lazy Loading*) usando imports dinâmicos.


5. Crie o mecanismo central em `src/app/router/router.js` que intercepta cliques com `data-link` e renderiza as páginas no elemento `#app` do `index.html`.



### Códigos de Referência Técnicos:

```javascript
// src/app/router/matchRoute.js
export function matchRoute(pathname, routePath) {
    const routeParts = routePath.split("/");
    const currentParts = pathname.split("/");
    if (routeParts.length !== currentParts.length) return null;
    const params = {};
    for (let i = 0; i < routeParts.length; i++) {
        if (routeParts[i].startsWith(":")) {
            params[routeParts[i].slice(1)] = currentParts[i];
            continue;
        }
        if (routeParts[i] !== currentParts[i]) return null;
    }
    return params;
}

```

```javascript
// src/app/router/router.js
import { routes } from "./routes";
import { matchRoute } from "./matchRoute";
import { navigate } from "./navigate";

class RouterClass {
    async render() {
        const pathname = location.pathname;
        const app = document.getElementById("app");
        
        for (const route of routes) {
            if (route.redirect && pathname === route.path) {
                navigate(route.redirect);
                return;
            }
            const params = matchRoute(pathname, route.path);
            if (!params) continue;

            // Placeholder para o Guard (implementado na Sprint 2)
            const token = localStorage.getItem("access_token");
            if (route.private && !token) {
                navigate("/login");
                return;
            }

            const module = await route.page();
            const html = await module.default(params);
            app.innerHTML = html;
            if (module.afterRender) module.afterRender(params);
            return;
        }
        // 404 fallback
        app.innerHTML = `<div class="p-8 text-center"><h1>404 - Página Não Encontrada</h1><a href="/hub" data-link>Voltar ao Hub</a></div>`;
    }

    init() {
        window.addEventListener("popstate", () => this.render());
        window.addEventListener("rerender", () => this.render());
        document.addEventListener("click", e => {
            const link = e.target.closest("[data-link]");
            if (!link) return;
            e.preventDefault();
            navigate(link.getAttribute("href"));
        });
        this.render();
    }
}
export const Router = new RouterClass();
```

## 🚀 SPRINT 2: Gerenciamento de Estado Global Reativo (Store)

**Objetivo:** Criar um fluxo previsível de estado único (estilo Redux/Zustand) que gerencie Usuário logado, Carrinho, Wishlist e Biblioteca, propagando atualizações visuais automaticamente.

### Instruções de Implementação:

1. Crie o arquivo `src/store/createStore.js` implementando o padrão de subscrição e listeners.


2. Crie o gerenciador local de persistência em `src/core/storage/storage.js` para evitar a perda do estado global em atualizações de página (F5).


3. Instancie o estado em `src/store/store.js` acoplado ao `Storage` de salvamento automático.


4. Centralize modificações de dados em `src/store/actions.js` disparando um evento global de `rerender` para o roteador.


5. Crie seletores limpos em `src/store/selectors.js` para leitura do estado.



### Códigos de Referência Técnicos:

```javascript
// src/store/createStore.js
export function createStore(initialState) {
    let state = initialState;
    const listeners = [];
    const getState = () => state;
    const setState = (updater) => {
        state = typeof updater === "function" ? updater(state) : updater;
        listeners.forEach(listener => listener(state));
    };
    const subscribe = (listener) => {
        listeners.push(listener);
        return () => {
            const index = listeners.indexOf(listener);
            if (index >= 0) listeners.splice(index, 1);
        };
    };
    return { getState, setState, subscribe };
}

```

```javascript
// src/store/actions.js
import { Store } from "./store";

export const Actions = {
    setUser(user) {
        Store.setState(state => ({ ...state, user }));
        window.dispatchEvent(new CustomEvent("rerender"));
    },
    logout() {
        localStorage.removeItem("access_token");
        Store.setState(state => ({ ...state, user: null, cart: [], wishlist: [], library: [] }));
        window.dispatchEvent(new CustomEvent("rerender"));
    },
    addToCart(game) {
        Store.setState(state => {
            if (state.cart.some(item => item.id === game.id)) return state;
            return { ...state, cart: [...state.cart, game] };
        });
        window.dispatchEvent(new CustomEvent("rerender"));
    },
    removeFromCart(gameId) {
        Store.setState(state => ({ ...state, cart: state.cart.filter(item => item.id !== gameId) }));
        window.dispatchEvent(new CustomEvent("rerender"));
    },
    toggleWishlist(game) {
        Store.setState(state => {
            const exists = state.wishlist.some(item => item.id === game.id);
            const wishlist = exists ? state.wishlist.filter(item => item.id !== game.id) : [...state.wishlist, game];
            return { ...state, wishlist };
        });
        window.dispatchEvent(new CustomEvent("rerender"));
    },
    purchaseCart() {
        Store.setState(state => ({
            ...state,
            library: [...state.library, ...state.cart],
            cart: []
        }));
        window.dispatchEvent(new CustomEvent("rerender"));
    }
};

```

```javascript
// src/pages/game/GamePage.js
import PublicLayout from "../../app/layouts/PublicLayout";
import { attachNavbarEvents } from "../../components/layout/Navbar";
import { GamesService } from "../../services/games/games.service";
import { Store } from "../../store/store";
import { Actions } from "../../store/actions";

export default async function GamePage(params) {
    const game = await GamesService.getBySlug(params.slug);
    if (!game) return PublicLayout(`<div class="p-12 text-center text-zinc-400">Jogo não encontrado.</div>`);

    const state = Store.getState();
    const isInCart = state.cart.some(item => item.id === game.id);
    const isInWishlist = state.wishlist.some(item => item.id === game.id);
    const isInLibrary = state.library.some(item => item.id === game.id);

    let actionsHtml = "";
    if (isInLibrary) {
        actionsHtml = `<span class="block text-center py-3 bg-zinc-800 text-green-400 font-bold rounded-lg border border-green-900/30">✓ Adquirido (Na sua Biblioteca)</span>`;
    } else {
        actionsHtml = `
            <button id="btn-add-cart" class="w-full py-3 ${isInCart ? "bg-zinc-800 text-zinc-400" : "bg-blue-600 hover:bg-blue-500 text-white"} font-bold rounded-lg transition" ${isInCart ? "disabled" : ""}>
                ${isInCart ? "Já no Carrinho" : "🛒 Adicionar ao Carrinho"}
            </button>
            <button id="btn-toggle-wish" class="w-full py-2.5 border ${isInWishlist ? "border-red-500 text-red-400 bg-red-950/10" : "border-zinc-700 text-zinc-300 hover:bg-zinc-900"} font-semibold rounded-lg transition">
                ${isInWishlist ? "❤️ Na Lista de Desejos" : "🤍 Adicionar à Lista de Desejos"}
            </button>
        `;
    }

    const screenshotsHtml = game.screenshots.map(src => `
        <div class="aspect-video bg-cover bg-center rounded-lg border border-zinc-800" style="background-image: url('${src}')"></div>
    `).join("");

    return PublicLayout(`
        <div class="flex flex-col gap-8">
            <div class="h-48 md:h-80 w-full bg-cover bg-center rounded-2xl relative" style="background-image: url('${game.banner_url}')">
                <div class="absolute inset-0 bg-gradient-to-t from-zinc-950 to-transparent"></div>
            </div>
            
            <div class="flex flex-col lg:flex-row gap-8 items-start">
                <div class="w-full lg:w-2/3 flex flex-col gap-6">
                    <div>
                        <h1 class="text-3xl md:text-5xl font-black text-white tracking-tight">${game.title}</h1>
                        <p class="text-sm text-blue-400 font-semibold mt-1">${game.publisher}</p>
                    </div>
                    <p class="text-zinc-300 leading-relaxed text-base">${game.long_description}</p>
                    
                    <div>
                        <h2 class="text-lg font-bold text-white mb-3">🖼 Capturas de Tela</h2>
                        <div class="grid grid-cols-1 sm:grid-cols-3 gap-4">${screenshotsHtml}</div>
                    </div>

                    <div class="bg-zinc-900 p-5 rounded-xl border border-zinc-800 grid grid-cols-1 md:grid-cols-2 gap-6 text-sm">
                        <div>
                            <h3 class="text-zinc-400 font-bold mb-2 uppercase tracking-wide text-xs">Requisitos Mínimos</h3>
                            <p><strong class="text-zinc-300">SO:</strong> ${game.requirements.minimum.os}</p>
                            <p><strong class="text-zinc-300">CPU:</strong> ${game.requirements.minimum.cpu}</p>
                            <p><strong class="text-zinc-300">RAM:</strong> ${game.requirements.minimum.ram}</p>
                            <p><strong class="text-zinc-300">GPU:</strong> ${game.requirements.minimum.gpu}</p>
                        </div>
                        <div>
                            <h3 class="text-zinc-400 font-bold mb-2 uppercase tracking-wide text-xs">Requisitos Recomendados</h3>
                            <p><strong class="text-zinc-300">SO:</strong> ${game.requirements.recommended.os}</p>
                            <p><strong class="text-zinc-300">CPU:</strong> ${game.requirements.recommended.cpu}</p>
                            <p><strong class="text-zinc-300">RAM:</strong> ${game.requirements.recommended.ram}</p>
                            <p><strong class="text-zinc-300">GPU:</strong> ${game.requirements.recommended.gpu}</p>
                        </div>
                    </div>
                </div>

                <div class="w-full lg:w-1/3 bg-zinc-900 p-6 rounded-xl border border-zinc-800 flex flex-col gap-4 sticky top-24">
                    <div class="flex flex-col">
                        <span class="text-xs text-zinc-400 uppercase font-bold tracking-wide">Preço Final</span>
                        <span class="text-3xl font-black text-blue-400">R$ ${game.price.toFixed(2)}</span>
                    </div>
                    <div class="flex flex-col gap-2 mt-2">${actionsHtml}</div>
                </div>
            </div>
        </div>
    `);
}

export async function afterRender(params) {
    attachNavbarEvents();
    const game = await GamesService.getBySlug(params.slug);
    if (!game) return;

    const cartBtn = document.getElementById("btn-add-cart");
    const wishBtn = document.getElementById("btn-toggle-wish");

    if (cartBtn) {
        cartBtn.addEventListener("click", () => {
            const token = localStorage.getItem("access_token");
            if (!token) {
                const { navigate } = require("../../app/router/navigate");
                navigate("/login");
                return;
            }
            Actions.addToCart(game);
        });
    }
    if (wishBtn) {
        wishBtn.addEventListener("click", () => {
            const token = localStorage.getItem("access_token");
            if (!token) {
                const { navigate } = require("../../app/router/navigate");
                navigate("/login");
                return;
            }
            Actions.toggleWishlist(game);
        });
    }
}
```

```javascript
// src/pages/cart/CartPage.js
import PrivateLayout from "../../app/layouts/PrivateLayout";
import PublicLayout from "../../app/layouts/PublicLayout";
import { attachNavbarEvents } from "../../components/layout/Navbar";
import { Store } from "../../store/store";
import { Actions } from "../../store/actions";

export default async function CartPage() {
    const state = Store.getState();
    const token = localStorage.getItem("access_token");
    
    if (!token) return PublicLayout(`<div class="p-12 text-center text-zinc-400">Por favor, faça login para ver seu carrinho.</div>`);

    if (state.cart.length === 0) {
        return PrivateLayout(`
            <div class="p-12 text-center bg-zinc-900 rounded-xl border border-zinc-800 text-zinc-400">
                <p class="text-lg font-semibold">Seu carrinho está vazio</p>
                <a href="/hub" class="text-blue-500 hover:underline mt-2 inline-block text-sm" data-link>Explorar a loja</a>
            </div>
        `);
    }

    const total = state.cart.reduce((acc, curr) => acc + curr.price, 0);

    const itemsHtml = state.cart.map(game => `
        <div class="flex items-center justify-between p-4 bg-zinc-900 border border-zinc-800 rounded-lg gap-4 flex-wrap">
            <div class="flex items-center gap-4">
                <div class="w-16 h-20 bg-cover bg-center rounded" style="background-image: url('${game.cover_url}')"></div>
                <div>
                    <h4 class="font-bold text-white text-base">${game.title}</h4>
                    <p class="text-xs text-zinc-400">${game.publisher}</p>
                </div>
            </div>
            <div class="flex items-center gap-6 ml-auto">
                <span class="font-bold text-blue-400 text-lg">R$ ${game.price.toFixed(2)}</span>
                <button data-id="${game.id}" class="btn-remove-item text-xs text-red-400 hover:text-red-300 font-medium">Remover</button>
            </div>
        </div>
    `).join("");

    return PrivateLayout(`
        <div class="flex flex-col gap-6">
            <h1 class="text-2xl font-black text-white tracking-tight">🛒 Meu Carrinho de Compras</h1>
            <div class="flex flex-col lg:flex-row gap-6 items-start">
                <div class="w-full lg:w-2/3 flex flex-col gap-4">${itemsHtml}</div>
                <div class="w-full lg:w-1/3 bg-zinc-900 border border-zinc-800 p-6 rounded-xl flex flex-col gap-4">
                    <h3 class="text-sm font-bold text-zinc-400 uppercase tracking-wide">Resumo do Pedido</h3>
                    <div class="flex justify-between items-center border-t border-zinc-800 pt-3">
                        <span class="text-zinc-300">Total Geral:</span>
                        <span class="text-2xl font-black text-blue-400">R$ ${total.toFixed(2)}</span>
                    </div>
                    <button id="btn-checkout" class="w-full py-3 bg-green-600 hover:bg-green-500 font-bold rounded-lg text-white text-center shadow transition mt-2">
                        Finalizar Compra
                    </button>
                </div>
            </div>
        </div>
    `);
}

export function afterRender() {
    attachNavbarEvents();
    document.querySelectorAll(".btn-remove-item").forEach(btn => {
        btn.addEventListener("click", (e) => {
            const id = e.target.getAttribute("data-id");
            Actions.removeFromCart(id);
        });
    });

    const checkoutBtn = document.getElementById("btn-checkout");
    if (checkoutBtn) {
        checkoutBtn.addEventListener("click", () => {
            Actions.purchaseCart();
            alert("Compra realizada com sucesso! Os jogos foram inseridos na sua biblioteca.");
            const { navigate } = require("../../app/router/navigate");
            navigate("/library");
        });
    }
}
```
